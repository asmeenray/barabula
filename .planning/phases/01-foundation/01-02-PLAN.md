```typescript
// API_BASE_URL must be imported from services/api.ts (now corrected to port 8000 by plan 01-01)
// Or use the same constant pattern — match whatever pattern authSlice uses for the API URL

export const fetchChatHistory = createAsyncThunk(
  'chat/fetchHistory',
  async (_, { getState, rejectWithValue }) => {
    try {
      const state = getState() as { auth: { token: string | null } };
      const token = state.auth.token;
      if (!token) throw new Error('No token');
      const response = await fetch('http://localhost:8000/api/v1/chat/history', {
        headers: { Authorization: `Bearer ${token}` },
      });
      if (!response.ok) throw new Error('Failed to fetch history');
      return await response.json();
    } catch (error: any) {
      return rejectWithValue(error.message);
    }
  }
);
```
