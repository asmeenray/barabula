   ```typescript
   useEffect(() => {
     if (!data?.activities) return
     const activities = data.activities.filter(a => a.location)
     Promise.all(
       activities.map(async (act, i) => {
         const coords = await resolveActivityCoordinates(act, data.destination)
         if (!coords) return null
         return {
           id: act.id,
           name: act.name,
           day: act.day_number,
           lng: coords.lng,
           lat: coords.lat,
           type: (act.activity_type === 'hotel' ? 'hotel' : 'activity') as 'activity' | 'hotel',
           sequenceNumber: i + 1,
         }
       })
     ).then(results => setMapPins(results.filter(Boolean) as MapPin[]))
   }, [data?.activities, data?.destination])
   ```
5. Handle pin click: `function handlePinClick(id: string) { setActiveActivityId(id); document.getElementById(`activity-${id}`)?.scrollIntoView({ behavior: 'smooth', block: 'center' }) }`
6. Handle card click: `function handleCardClick(id: string) { setActiveActivityId(id) }`
7. Handle day change from DayPillNav: `const [activeDay, setActiveDay] = useState<number | null>(null)` — when pill clicked, update activeDay; pass to both DayPillNav and ItineraryMap.
